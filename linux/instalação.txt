# Instalando JRE e JDK
sudo apt-get remove openjdk* (Remove o OpenJDK)
sudo apt-get update
sudo add-apt-repository ppa:webupd8team/java
sudo apt-get update
sudo apt-get install oracle-java8-installer (Instala JRE/JDK)
java - version
javac - version
sudo update-alternatives --config java (Escolher versão DEFAULT do JAVA instaladas)
sudo update-alternatives --config javac

#Configurando variável JAVA_HOME:
sudo update-alternatives --config java (Pegar o camingo do Java instalado)
sudo nano /etc/enviroment (Abrir o arquivo de configuração)
JAVA_HOME="/usr/lib/jvm/java-8-oracle/jre/bin/"  (Colocar no final do arquivo, salvar e sair)
source /etc/environment (Atualizar mudança realizada)
